public class CuentaBancaria {
private string titular;
private string numeroCuenta;
private double saldo;
ppublic CuentaBancaria() {
titular = "Sin nombre";
numeroCuenta = "000000";
saldo = 0;
}

public CuentaBancaria (string t, string n, double s) {
if (t.isEmpy()) {
titular = "sin nombre";
} else {
titular = t;
}

if (n.lenght() < 6 {
numeroCuenta = "000000"
} else {
numeroCuenta = n;
}

if (s < 0) {
saldo = 0 
} else {
saldo = s;
}
}
public void depositar(double monto) {
if (monto > 0) {
saldo = saldo + monto;
}
}
public void retirar(double monto) {
if (monto > 0 && monto <= saldo) {
saldo = saldo - monto;
}
}
public string toString() {
return "itular: " + titular + "\nCuenta: " + numeroCuenta + "\nSaldo: " + saldo;
}
}
